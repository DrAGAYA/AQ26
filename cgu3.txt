    function selectService(id, name, price) {
        document.querySelectorAll('.service-option').forEach(opt => opt.classList.remove('selected'));
        document.getElementById(id).classList.add('selected');
        
        document.getElementById('selectedServiceInput').value = name;
        document.getElementById('selectedPriceInput').value = price;
        
        document.getElementById('checkoutService').innerText = name;
        document.getElementById('checkoutPrice').innerText = price;

    function goStep(n) {
        document.querySelectorAll('.step').forEach(s => s.classList.remove('active'));
        document.querySelectorAll('.tooltip-box').forEach(b => b.classList.remove('show'));

        const header = document.getElementById('mainHeader');

        if (n === 'garde') {
            document.getElementById('stepGarde').classList.add('active');
            header.style.display = "none";
        } else if (n === 'accueil') {
            document.getElementById('stepAccueil').classList.add('active');
            header.style.display = "none";
        } else if (n === 'catalogue') {
            document.getElementById('stepCatalogue').classList.add('active');
            header.style.display = "none";
        } else if (n === 'paiement') {
            document.getElementById('stepPaiement').classList.add('active');
            header.style.display = "none";
        } else {
            document.getElementById('step' + n).classList.add('active');
            if (n > 0) {
                header.style.display = "block";
                document.getElementById('progress-fill').style.width = (n * 10) + "%";
                document.getElementById('current-num').innerText = n;
            } else {
                header.style.display = "none";
            }
        }
        window.scrollTo(0,0);
    }

    function validateStepAccueil() {
        const form = document.getElementById('quizForm');
        if (form.elements['Nom_Utilisateur'].reportValidity() && 
            form.elements['_replyto'].reportValidity() && 
            form.querySelector('input[type="checkbox"]').reportValidity()) {
            goStep('catalogue');
        }
    }

    function toggleTooltip(id) {
        const tooltip = document.getElementById('tip' + id);
        document.querySelectorAll('.tooltip-box').forEach(box => {
            if (box.id !== 'tip' + id) box.classList.remove('show');
        });
        tooltip.classList.toggle('show');
    }
